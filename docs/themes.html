Usage Example
import { useTheme } from 'react-native-heroui';

function MyComponent() {
  const theme = useTheme();
  
  return (
    <View style={{ backgroundColor: theme.colors['primary-500'] }}>
      <Text style={{ color: theme.colors.foreground }}>
        Themed Text
      </Text>
    </View>
  );
}

function MyComponent() {
  const theme = useTheme();
  
  return (
    <Text style={{

    }}>

    </Text>
  );
}

Spacing

Unit-based spacing system with a 4px base unit for consistent spacing throughout your app.

Usage Example
import { useTheme } from 'react-native-heroui';

function MyComponent() {
  const theme = useTheme();
  
  return (
    <View style={{
      padding: theme.spacing['unit-4'],
      marginBottom: theme.spacing['unit-md'],
      gap: theme.spacing['unit-2'],
    }}>
      {/* Content */}
    </View>
  );
}
